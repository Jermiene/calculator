Grade Calculator

from pyscript import Element def calculate_grade(prelim_grade): prelim_weight = 0.20 midterm_weight = 0.30 finals_weight = 0.50 target_grade = 75 deans_list_grade = 90 prelim_grade = float(prelim_grade) required_from_midterm_finals = target_grade - (prelim_grade * prelim_weight) combined_grade = required_from_midterm_finals / (midterm_weight + finals_weight) if combined_grade < 0 or combined_grade > 100: return "Sorry, it's impossible to reach the target grade with the given prelim grade." deans_required = deans_list_grade - (prelim_grade * prelim_weight) deans_combined_grade = deans_required / (midterm_weight + finals_weight) if deans_combined_grade > 100: deans_message = "Unfortunately, it's not possible to achieve a 90 overall grade based on your Prelim grade." else: deans_message = f"To achieve a 90 overall grade (Dean's Lister), you need to get at least {deans_combined_grade:.2f} in both Midterm and Final." result_message = f"With a Prelim grade of {prelim_grade:.2f}, to get an overall grade of at least 75, you need to get at least {combined_grade:.2f} in both Midterm and Final.
{deans_message}" return result_message def calculateGrade(event): prelim_grade = Element("prelim_grade").element.value result = calculate_grade(prelim_grade) Element("output").element.innerHTML = result # Bind the function to the button click event Element("calculate-button").element.onclick = calculateGrade
